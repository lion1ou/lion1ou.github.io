2016-10-19 /

Express жҳҜдёҖз§ҚдҝқжҢҒжңҖдҪҺзЁӢеәҰи§„жЁЎзҡ„зҒөжҙ» Node.js Web еә”з”ЁзЁӢеәҸжЎҶжһ¶пјҢдёә Web е’Ңз§»еҠЁеә”з”ЁзЁӢеәҸжҸҗдҫӣдёҖз»„ејәеӨ§зҡ„еҠҹиғҪгҖӮиҜҰи§Ғпјҡе®ҳзҪ‘ гҖҒдёӯж–ҮзҪ‘з«ҷ

дёҖгҖҒеҮҶеӨҮе·ҘдҪң

йҰ–е…ҲдҪ йңҖиҰҒе®үиЈ…NodeJSзҺҜеўғ иҝҷйҮҢдёҚеҶҚеҒҡд»Ӣз»ҚгҖӮ

1.1 е®үиЈ…Express

  npm install express -g //е®үиЈ…е…ЁеұҖзҡ„express
  npm install express-generator -g //е®үиЈ…е…ЁеұҖзҡ„expressе‘Ҫд»ӨиЎҢе·Ҙе…·

1.2 еҲқе§ӢеҢ–йЎ№зӣ®

  cd code //йЎ№зӣ®з©әй—ҙ
  express todo (-e) //йЎ№зӣ®еҗҚз§°пјҲж·»еҠ еҸӮж•°жҳҜдҪҝз”ЁejsжЁЎжқҝпјҢй»ҳи®ӨдҪҝз”ЁjadeпјүпјҢз”ҹжҲҗдёҖдёӘexpressи„ҡжүӢжһ¶йЎ№зӣ®

йЎ№зӣ®зӣ®еҪ•еҰӮдёӢпјҡ

todo

        error.ejs

    package.json     //йЎ№зӣ®зҡ„й…ҚзҪ®ж–Үд»¶

1.5 еҲқе§ӢеҢ–йЎ№зӣ®

ж №жҚ®з»Ҳз«ҜжңҖеҗҺдёӨиЎҢжҸҗзӨәпјҡ

жү§иЎҢдёӢйқўд»Јз ҒпјҢеҲқе§ӢеҢ–йЎ№зӣ®пјҢе®үиЈ…жүҖжңүдҫқиө–гҖӮ

cd todo && npm install //иҝӣе…ҘtodoйЎ№зӣ®зӣ®еҪ•пјҢе®үиЈ…зӣёе…ідҫқиө–

еңЁж–Үд»¶зӣ®еҪ•дёӯдјҡеӨҡеҮәдёҖдёӘж–Үд»¶еӨ№node_modules еҚідҫқиө–жЁЎеқ—зҡ„ж–Үд»¶еӨ№гҖӮ

дәҢгҖҒеҗҜеҠЁжңҚеҠЎ

2.1 еҗҜеҠЁжңҚеҠЎеҷЁ

npm start   //еҗҜеҠЁжңҚеҠЎеҷЁ

жҳҫзӨәдёҠйқўеӣҫзүҮпјҢиЎЁзӨәжңҚеҠЎеҷЁе·ІеҗҜеҠЁгҖӮ

2.2 жөҸи§ҲеҷЁи®ҝй—®

жү“ејҖжөҸи§ҲеҷЁпјҢеңЁжөҸи§ҲеҷЁдёӯи®ҝй—® http://localhost:3000/жҲ–http://127.0.0.1:3000/ жөҸи§ҲеҷЁиҫ“еҮәпјҡ

дёүгҖҒеҹәжң¬дҪҝз”Ё

д»ҘдёӢжҳҜapp.jsзҡ„зӣёе…ід»Јз Ғи§ЈйҮҠпјҢеҰӮжңүд»Җд№ҲдёҚеҜ№пјҢиҜ·жү№иҜ„жҢҮжӯЈгҖӮ

3.1 е®һзҺ°йҖ»иҫ‘

// еј•е…ҘеҜ№иұЎ
var express = require('express');
var path = require('path'); //pathеҜ№иұЎпјҢи§„иҢғиҝһжҺҘе’Ңи§Јжһҗи·Ҝеҫ„
var favicon = require('serve-favicon'); //йЎөйқўеӣҫж Ү
var logger = require('morgan'); //httpиҜ·жұӮж—Ҙеҝ—и®°еҪ•еҷЁ
var cookieParser = require('cookie-parser'); //и§Јжһҗcookie
var bodyParser = require('body-parser'); //еӨ„зҗҶиҜ·жұӮbodyзҡ„дёӯй—ҙд»¶
var app = express();
var routes = require('./routes/index'); //home pageжҺҘеҸЈ
var users = require('./routes/users'); //з”ЁжҲ·жҺҘеҸЈ

///======= view engine setup  жЁЎжқҝ ејҖе§Ӣ===========//
app.set('views', path.join(__dirname, 'views'));
app.set('view engine', 'ejs');
///======= view engine setup  жЁЎжқҝ з»“жқҹ===========//

///======= дҪҝз”Ёдёӯй—ҙд»¶ ејҖе§Ӣ===========//
// uncomment after placing your favicon in /public (йңҖиҰҒжҳҫзӨәfaviconж—¶пјҢеҸ–ж¶ҲжіЁйҮҠ)
//app.use(favicon(path.join(__dirname, 'public', 'favicon.ico')));

app.use(logger('dev')); //д»ҘејҖеҸ‘иҖ…жЁЎејҸпјҢдҪҝз”Ёlog
app.use(bodyParser.urlencoded({ extended: false })); //и§Јжһҗ application/x-www-form-urlencodedзұ»еһӢ
app.use(bodyParser.json()); //и§Јжһҗ application/jsonзұ»еһӢ
app.use(cookieParser()); //cookieдёӯй—ҙд»¶
app.use(express.static(path.join(__dirname, 'public'))); //йқҷжҖҒж–Үд»¶еӨ№ж–№жі•
///======= дҪҝз”Ёдёӯй—ҙд»¶ з»“жқҹ===========//


///=======и·Ҝз”ұдҝЎжҒҜ пјҲжҺҘеҸЈең°еқҖпјүејҖе§Ӣ ===========//
//еӯҳж”ҫеңЁ./routesзӣ®еҪ•дёӢ
app.use('/', routes); //еңЁappдёӯжіЁеҶҢroutesиҜҘжҺҘеҸЈ 
app.use('/users', users); //еңЁappдёӯжіЁеҶҢusersжҺҘеҸЈ
///=======и·Ҝз”ұдҝЎжҒҜ пјҲжҺҘеҸЈең°еқҖ) д»Ӣз»Қ===========//

// catch 404 and forward to error handler(жҚ•иҺ·404й”ҷиҜҜпјҢе№¶иҪ¬еҗ‘й”ҷиҜҜеӨ„зҗҶзЁӢеәҸ)
app.use(function(req, res, next) {
    var err = new Error('Not Found');
    err.status = 404;
    next(err);
});

///======= error handlers  й”ҷиҜҜеӨ„зҗҶзЁӢеәҸ  ејҖе§Ӣ========//

// development error handler(ејҖеҸ‘иҖ…й”ҷиҜҜеӨ„зҗҶзЁӢеәҸ)
// will print stacktrace(е°Ҷдјҡжү“еҚ°е Ҷж Ҳи·ҹиёӘ)
if (app.get('env') === 'development') {
    app.use(function(err, req, res, next) {
        res.status(err.status || 500); //иҝ”еӣһй”ҷиҜҜзҠ¶жҖҒз Ғ
        res.render('error', { //жёІжҹ“errorи§ҶеӣҫжЁЎжқҝ
            message: err.message,
            error: err
        });
    });
}
// production error handler(з”ҹдә§зҺҜеўғдёӢзҡ„еӨ„зҗҶзЁӢеәҸ)
// no stacktraces leaked to user(дёҚдјҡжі„йңІе Ҷж Ҳи·ҹиёӘз»ҷз”ЁжҲ·)
app.use(function(err, req, res, next) {
    res.status(err.status || 500);
    res.render('error', {
        message: err.message,
        error: {}
    });
});
///======= error handlers  й”ҷиҜҜеӨ„зҗҶзЁӢеәҸ  з»“жқҹ========//
module.exports = app;

еҪ“жҲ‘д»¬еңЁжөҸи§ҲеҷЁдёӯ и®ҝй—® http://localhost:3000/ ж—¶йҖҡиҝҮapp.jsдёӯзҡ„д»Јз Ғпјҡ

app.use('/', routes); //еңЁappдёӯжіЁеҶҢroutesиҜҘжҺҘеҸЈ

и°ғз”Ё routes/index.js пјҢеҚіпјҡ

var express = require('express');
var router = express.Router();//дҪҝз”Ё express.Router зұ»еҲӣе»әжЁЎеқ—еҢ–гҖҒеҸҜжҢӮиҪҪзҡ„и·Ҝз”ұеҸҘжҹ„

//е®ҡд№үдёҖдёӘgetиҜ·жұӮ pathдёәж №зӣ®еҪ•'/'
/* GET home page. */
router.get('/', function(req, res, next) {
  res.render('index', { title: 'Express' });//жёІжҹ“indexи§ҶеӣҫжЁЎжқҝгҖӮ
});

module.exports = router;//жҡҙйңІеӨ–йғЁжҺҘеҸЈrouter

жүҖд»ҘеҪ“жҲ‘д»¬еңЁжөҸи§ҲеҷЁдёӯ и®ҝй—® http://localhost:3000/users ж—¶йҖҡиҝҮapp.jsдёӯзҡ„д»Јз Ғпјҡapp.use('/users', users); //еңЁappдёӯжіЁеҶҢroutesиҜҘжҺҘеҸЈ
и°ғз”Ё routes/users.js

3.2 и·Ҝз”ұе®ҡд№ү

е®ҡд№үдёҖдёӘи·Ҝз”ұзҡ„еҹәжң¬ж јејҸдёәпјҡ

app.METHOD(PATH, HANDLER)//PATH жҳҜжңҚеҠЎеҷЁдёҠзҡ„и·Ҝеҫ„гҖӮHANDLER жҳҜеңЁи·Ҝз”ұеҢ№й…Қж—¶жү§иЎҢзҡ„еҮҪж•°гҖӮ

METHODжҳҜHTTP иҜ·жұӮеҜ№еә”зҡ„и·Ҝз”ұж–№жі•пјҡget, post, put, head, deleteзӯүгҖӮе…¶дёӯapp.all()жҳҜдёҖдёӘзү№ж®Ҡзҡ„и·Ҝз”ұж–№жі•пјҢжІЎжңүд»»дҪ• HTTP ж–№жі•дёҺе…¶еҜ№еә”пјҢе®ғзҡ„дҪңз”ЁжҳҜеҜ№дәҺдёҖдёӘи·Ҝеҫ„дёҠзҡ„жүҖжңүиҜ·жұӮеҠ иҪҪдёӯй—ҙд»¶гҖӮ

3.3 е“Қеә”ж–№жі•

ж–№жі• жҸҸиҝ°
res.download() жҸҗзӨәдёӢиҪҪж–Үд»¶гҖӮ
res.end() з»Ҳз»“е“Қеә”еӨ„зҗҶжөҒзЁӢгҖӮ
res.json() еҸ‘йҖҒдёҖдёӘ JSON ж јејҸзҡ„е“Қеә”гҖӮ
res.jsonp() еҸ‘йҖҒдёҖдёӘж”ҜжҢҒ JSONP зҡ„ JSON ж јејҸзҡ„е“Қеә”гҖӮ
res.redirect() йҮҚе®ҡеҗ‘иҜ·жұӮгҖӮ
res.render() жёІжҹ“и§ҶеӣҫжЁЎжқҝгҖӮ
res.send() еҸ‘йҖҒеҗ„з§Қзұ»еһӢзҡ„е“Қеә”гҖӮ
res.sendFile д»Ҙе…«дҪҚеӯ—иҠӮжөҒзҡ„еҪўејҸеҸ‘йҖҒж–Үд»¶гҖӮ
res.sendStatus() и®ҫзҪ®е“Қеә”зҠ¶жҖҒд»Јз ҒпјҢе№¶е°Ҷе…¶д»Ҙеӯ—з¬ҰдёІеҪўејҸдҪңдёәе“Қеә”дҪ“зҡ„дёҖйғЁеҲҶеҸ‘йҖҒгҖӮ

3.4 жҺҘеҸЈе®һдҫӢ

дёӢйқўжҲ‘д»¬жқҘе®һзҺ°дёҖдёӘиҺ·еҸ–з”ЁжҲ·дҝЎжҒҜжҺҘеҸЈпјҡ

3.4.1 еҲӣе»әжЁЎеһӢ

еңЁroutesж–Үд»¶еӨ№еҶ…еҲӣе»әmodel.jsпјҢе®ҡд№үдёҖдёӘUserжЁЎеһӢгҖӮ

function User() {
    this.name;
    this.city;
    this.age;
}
module.exports = User;

3.4.2 ж·»еҠ ж–№жі•

еңЁroutes/users.jsж–Үд»¶дёӯж·»еҠ еј•з”Ёпјҡ

var URL = require('url'); //еј•е…ҘURLдёӯй—ҙд»¶
var User = require('./model'); //еј•е…Ҙж•°жҚ®жЁЎеһӢ

е№¶з»§з»ӯж·»еҠ еҰӮдёӢж–№жі•пјҡ

router.get('/getUserInfo', function(req, res, next) {
    var user = new User(); //е®һдҫӢеҢ–еҜ№иұЎ
    var params = URL.parse(req.url, true).query;//иҺ·еҸ–urlеҸӮж•°
    console.log('params', params);
    if (params.id == '1') { //ж №жҚ®еҸӮж•°иөӢеҖј
        user.name = "luo";
        user.age = "11";
        user.city = "еҢ—дә¬еёӮ";
    } else {
        user.name = "lion";
        user.age = "22";
        user.city = "жқӯе·һеёӮ";
    }
    //иҝҷйҮҢжҲ‘д»¬еҶҷжӯ»дәҶиҝ”еӣһж•°жҚ®пјҢе№¶жІЎжңүжҹҘиҜўж•°жҚ®еә“гҖӮ
    var response = { status: 200, data: user }; //е®ҡд№үиҝ”еӣһж•°жҚ®
    res.send(JSON.stringify(response)); //еҸ‘йҖҒиҜ·жұӮ
});
